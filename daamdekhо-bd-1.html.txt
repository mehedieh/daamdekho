function setupSearch(inputId, ddId, clearId) {
  const input = document.getElementById(inputId);
  const dd = document.getElementById(ddId);
  const clearBtn = document.getElementById(clearId);

  input.addEventListener('input', () => {
    clearBtn.classList.toggle('visible', input.value.length > 0);
    clearTimeout(debounceTimer);
    debounceTimer = setTimeout(() => renderDropdown(input.value, dd), 150);
    dd.classList.add('open');
  });

  input.addEventListener('focus', () => {
    renderDropdown(input.value, dd);
    dd.classList.add('open');
  });

  input.addEventListener('keydown', e => {
    if(e.key === 'Enter') { doSearchFromInput(input.value); dd.classList.remove('open'); }
    if(e.key === 'Escape') dd.classList.remove('open');
  });

  clearBtn.addEventListener('click', () => {
    input.value = '';
    clearBtn.classList.remove('visible');
    dd.classList.remove('open');
  });

  document.addEventListener('click', e => {
    if(!input.closest('.search-wrap').contains(e.target)) dd.classList.remove('open');
  });
}

// ── SerpApi fetch ──
// NOTE: Direct browser → SerpApi calls are possible but expose your key.
// For production, route through your own backend: /api/search?q=...
// The CORS proxy below (allorigins) is for development/demo only.
async function fetchFromSerpApi(query) {
  const key = getSerpKey();
  if(!key) return null;

  // Option A: Direct call (key visible in DevTools — dev/demo only)
  // Option B: Via a proxy (used here for demo)
  // Option C (recommended for prod): your own Next.js API route
  const serpUrl = `https://serpapi.com/search.json?engine=google_shopping&q=${encodeURIComponent(query)}&gl=bd&hl=en&num=20&api_key=${key}`;
  const proxyUrl = `https://api.allorigins.win/get?url=${encodeURIComponent(serpUrl)}`;

  try {
    const res = await fetch(proxyUrl);
    if(!res.ok) throw new Error(`HTTP ${res.status}`);
    const outer = await res.json();
    const data = JSON.parse(outer.contents);

    if(data.error) throw new Error(data.error);

    return normalizeSerpResults(data, query);
  } catch(err) {
    console.error('SerpApi error:', err);
    toast(`⚠️ SerpApi error: ${err.message.substring(0,60)}`);
    return null;
  }
}

// ── Normalize SerpApi Google Shopping response → our product schema ──
function normalizeSerpResults(data, query) {
  const items = data.shopping_results || [];
  if(!items.length) return [];

  return items.slice(0, 20).map((item, i) => {
    // Parse price — SerpApi returns e.g. "৳12,999" or "BDT 12999" or just "12999"
    const priceRaw = item.price || item.extracted_price || '0';
    const price = parsePrice(priceRaw);

    // Try to detect old/original price
    const oldPriceRaw = item.old_price || item.original_price || null;
    const oldPrice = oldPriceRaw ? parsePrice(oldPriceRaw) : null;

    // Source / store name — SerpApi gives source field
    const storeName = cleanStoreName(item.source || 'Online Store');

    // Rating
    const rating = item.rating ? parseFloat(item.rating) : null;

    // Delivery
    const delivery = item.delivery ? parseDelivery(item.delivery) : 'N/A';

    return {
      id: item.product_id || item.position?.toString() || `serp_${i}`,
      title: item.title || query,
      image: item.thumbnail || item.image || '',
      store: storeName,
      price,
      oldPrice: oldPrice && oldPrice > price ? oldPrice : null,
      rating: rating || +(3.5 + Math.random() * 1.2).toFixed(1),
      delivery,
      link: item.link || item.product_link || '#',
      category: item.category || 'General',
      brand: extractBrand(item.title || query),
      source: 'serpapi', // mark as live data
    };
  }).filter(p => p.price > 0); // drop items with unparseable price
}

// ── Price parser — handles ৳12,999 / BDT 12,999 / $12.99 / 12999 ──
function parsePrice(raw) {
  if(typeof raw === 'number') return Math.round(raw);
  // Remove currency symbols and commas
  const cleaned = String(raw).replace(/[৳$£€,\s]/g,'').replace(/BDT/i,'').replace(/USD/i,'').trim();
  const num = parseFloat(cleaned);
  if(isNaN(num)) return 0;
  // If price looks like USD (small number), convert to BDT (approx 110 BDT/USD)
  if(num < 500 && String(raw).includes('$')) return Math.round(num * 110);
  return Math.round(num);
}

// ── Delivery parser ──
function parseDelivery(raw) {
  if(!raw) return 'N/A';
  const s = String(raw).toLowerCase();
  if(s.includes('free')) return 'Free';
  const match = s.match(/[\d,]+/);
  if(match) return `৳${match[0]}`;
  return raw.substring(0, 20);
}

// ── Store name cleaner — maps common BD store variants ──
const STORE_MAP = {
  'daraz': 'Daraz',
  'pickaboo': 'Pickaboo',
  'ajkerdeal': 'AjkerDeal',
  'ryans': 'Ryans',
  'ryan': 'Ryans',
  'startech': 'StarTech',
  'star tech': 'StarTech',
  'chaldal': 'Chaldal',
  'shajgoj': 'Shajgoj',
  'othoba': 'Othoba',
  'rokomari': 'Rokomari',
  'amazon': 'Amazon',
  'flipkart': 'Flipkart',
};

function cleanStoreName(raw) {
  const lower = raw.toLowerCase();
  for(const [key, val] of Object.entries(STORE_MAP)) {
    if(lower.includes(key)) return val;
  }
  // Return cleaned version of raw
  return raw.replace(/\.com\.bd|\.com|\.bd/gi,'').replace(/\b\w/g,c=>c.toUpperCase()).trim().substring(0,20);
}

// ── Brand extractor ──
function extractBrand(title) {
  const brands = ['Apple','Samsung','Xiaomi','Realme','OnePlus','Oppo','Vivo','Sony','JBL','Asus','Dell','HP','Lenovo','Nokia','Motorola','Huawei','Honor','Google','Nothing'];
  for(const b of brands) {
    if(title.toLowerCase().includes(b.toLowerCase())) return b;
  }
  return title.split(' ')[0];
}

// ── Dynamic store color for unknown stores ──
function getStoreColor(storeName) {
  const known = STORES[storeName];
  if(known) return known;
  // Generate consistent color from store name
  let hash = 0;
  for(let c of storeName) hash = c.charCodeAt(0) + ((hash<<5)-hash);
  const h = Math.abs(hash) % 360;
  return { color: `hsl(${h},65%,45%)`, emoji: '🏪' };
}

async function fetchResults(query) {

  const hasKey = !!getSerpKey();

    toast('🔍 Fetching live prices from SerpApi…');
    results = await fetchFromSerpApi(query);
  }

  const useFallback = document.getElementById('useMockFallback')?.checked ?? true;

  if(!results || results.length === 0) {
    if(hasKey && results !== null) toast('⚠️ No live results — using mock data');
    if(useFallback || !hasKey) {
      results = getMockResults(query);
    } else {
      results = [];
    }
  } else {
    const count = results.length;
    toast(`✅ Found ${count} live results for "${query}"`);
  }

  allResults = results;
  hideLoading();
  activeStoreFilter = 'all';
  activePriceMin = null; activePriceMax = null; activeMinRating = null; freeDeliveryOnly = false;
  activeSort = 'price-asc';
  filteredResults = [...allResults];
  applyFiltersAndSort();
  document.getElementById('trendingSection').style.display = 'none';
  document.getElementById('filtersSection').style.display = 'block';
  document.getElementById('chartSection').style.display = 'block';
  document.getElementById('chartTitle').textContent = `Price History — ${query.charAt(0).toUpperCase()+query.slice(1)}`;
  renderChart(currentPeriod);
  updateAlertBtn();
  document.querySelectorAll('.filter-chip').forEach(c => c.classList.remove('active'));
  document.getElementById('filterAllBtn').classList.add('active');
}

function applyFiltersAndSort() {
  filteredResults = allResults.filter(p => {
    if(activeStoreFilter !== 'all' && p.store !== activeStoreFilter) return false;
    if(activePriceMin !== null && p.price < activePriceMin) return false;
    if(activePriceMax !== null && p.price > activePriceMax) return false;
    if(activeMinRating !== null && p.rating < activeMinRating) return false;
    if(freeDeliveryOnly && p.delivery !== 'Free') return false;
    return true;
  });

  filteredResults.sort((a, b) => {
    if(activeSort === 'price-asc') return a.price - b.price;
    if(activeSort === 'price-desc') return b.price - a.price;
    if(activeSort === 'rating') return b.rating - a.rating;
    if(activeSort === 'discount') {
      const discA = a.oldPrice ? a.oldPrice - a.price : 0;
      const discB = b.oldPrice ? b.oldPrice - b.price : 0;
      return discB - discA;
    }
    return 0;
  });

  renderResults();
}

/* ──────────────────────────────────────────────────────────
   WISHLIST
   ────────────────────────────────────────────────────────── */
function toggleWishItem(e, id) {
  e.stopPropagation();
  const p = allResults.find(p=>p.id===id);
  if(!p) return;
  const idx = wishlist.findIndex(w=>w.id===id);
  if(idx === -1) {
    wishlist.push(p);
    toast('❤️ Added to wishlist');
  } else {
    wishlist.splice(idx,1);
    toast('Removed from wishlist');
  }
  localStorage.setItem('dd_wishlist', JSON.stringify(wishlist));
  updateWishCount();
  renderResults();
  renderWishlist();
}

function updateWishCount() {
  const el = document.getElementById('wishCount');
  el.textContent = wishlist.length;
  el.style.display = wishlist.length ? 'flex' : 'none';
}

function toggleWishlist() {
  const panel = document.getElementById('wishlistPanel');
  const overlay = document.getElementById('wishlistOverlay');
  const open = panel.classList.toggle('open');
  overlay.style.display = open ? 'block' : 'none';
  if(open) renderWishlist();

function removeWish(id) {

/* ──────────────────────────────────────────────────────────
   COMPARE MODAL
   ────────────────────────────────────────────────────────── */
function addToCompare(id) {
  const p = allResults.find(p=>p.id===id);
  if(!p) return;
  if(compareList.find(c=>c.id===id)) { toast('Already in comparison'); return; }
  if(compareList.length >= 4) { toast('Max 4 items to compare'); return; }
  compareList.push(p);
  toast(`Added to compare (${compareList.length}/4)`);
}

function toggleCompareModal() {
  const modal = document.getElementById('compareModal');
  const open = modal.classList.toggle('open');
  if(open) renderCompare();
}

function renderCompare() {
  const body = document.getElementById('compareBody');
  if(!compareList.length) {
    // Show all current results
    const pool = filteredResults.length ? filteredResults : allResults;
    if(!pool.length) {
      body.innerHTML = `<div class="wishlist-empty"><div style="font-size:40px;margin-bottom:12px">🔍</div><p>Search for a product first, then click the compare button on cards.</p></div>`;
      return;
    }
    renderCompareCards(body, pool.slice(0,5));
    return;
  }
  renderCompareCards(body, compareList);
}

function renderCompareCards(body, items) {
  const minPrice = Math.min(...items.map(p=>p.price));
  const rows = [
    ['Price', p => `<strong>৳${p.price.toLocaleString()}</strong>`],
    ['Rating', p => `⭐ ${p.rating}`],
    ['Delivery', p => p.delivery],
    ['Original', p => p.oldPrice ? `৳${p.oldPrice.toLocaleString()}` : '—'],
    ['Savings', p => p.oldPrice ? `৳${(p.oldPrice-p.price).toLocaleString()}` : '—'],
  ];

  body.innerHTML = `
    <div class="compare-grid">
      ${items.map(p => {
        const isWinner = p.price === minPrice;
        const store = getStoreColor(p.store);
        return `<div class="compare-card${isWinner?' winner':''}">
          ${isWinner?'<div class="compare-badge">🟢 Best Deal</div>':''}
          <div class="compare-store" style="color:${store.color}">${p.store}</div>
          <div class="compare-price">৳${p.price.toLocaleString()}</div>
          <div style="font-size:11px;color:var(--text2);margin-top:4px;margin-bottom:12px;white-space:nowrap;overflow:hidden;text-overflow:ellipsis">${p.title.substring(0,30)}…</div>
          ${rows.map(([label, fn]) => `
            <div class="compare-row">
              <span class="compare-label">${label}</span>
              <span class="compare-val">${fn(p)}</span>
            </div>
          `).join('')}
          <button class="btn btn-primary" style="margin-top:12px;width:100%;font-size:12px" onclick="toast('Opening ${p.store}…')">Buy Now</button>
        </div>`;
      }).join('')}
    </div>
    <div style="text-align:center;margin-top:20px">
      <button class="btn btn-outline" onclick="compareList=[];renderCompare()">Clear Comparison</button>
    </div>
  `;
}

/* ──────────────────────────────────────────────────────────
   PRICE HISTORY CHART
   ────────────────────────────────────────────────────────── */
function generatePriceHistory(period) {
  const points = {' 1W':7,'1M':30,'3M':90,'1Y':365}[period] || 7;
  const base = allResults.length ? Math.min(...allResults.map(p=>p.price)) : 20000;
  const labels = [], data = [];
  const now = new Date();
  for(let i=points;i>=0;i--) {
    const d = new Date(now); d.setDate(now.getDate()-i);
    labels.push(d.toLocaleDateString('en-BD',{month:'short',day:'numeric'}));
    data.push(Math.round(base * (0.92 + Math.random() * 0.16)));
  }
  return {labels, data};
}

function renderChart(period) {
  const ctx = document.getElementById('priceChart').getContext('2d');
  const {labels, data} = generatePriceHistory(period);
  const isDark = document.documentElement.getAttribute('data-theme')==='dark';
  const accent = getComputedStyle(document.documentElement).getPropertyValue('--accent').trim() || '#3b5bdb';

  if(chartInstance) chartInstance.destroy();
  chartInstance = new Chart(ctx, {
    type:'line',
    data:{
      labels,
      datasets:[{
        label:'Lowest Price (৳)',
        data,
        borderColor: '#3b5bdb',
        backgroundColor: 'rgba(59,91,219,0.08)',
        tension:0.4,
        fill:true,
        pointRadius:3,
        pointHoverRadius:6,
        pointBackgroundColor:'#3b5bdb',
        borderWidth:2,
      }]
    },
    options:{
      responsive:true,
      maintainAspectRatio:false,
      plugins:{
        legend:{display:false},
        tooltip:{
          backgroundColor: isDark ? '#1c2333' : '#fff',
          borderColor: isDark ? '#2a3347' : '#e2e8f8',
          borderWidth:1,
          titleColor: isDark ? '#e8edf8' : '#0f1729',
          bodyColor: isDark ? '#8899bb' : '#5a6a8a',
          callbacks:{label:ctx=>`৳${ctx.parsed.y.toLocaleString()}`},
        }
      },
      scales:{
        x:{grid:{display:false},ticks:{color: isDark?'#8899bb':'#5a6a8a',maxTicksLimit:8,font:{size:11}}},
        y:{grid:{color: isDark?'rgba(255,255,255,0.05)':'rgba(0,0,0,0.04)'},ticks:{color: isDark?'#8899bb':'#5a6a8a',font:{size:11},callback:v=>`৳${v.toLocaleString()}`}},
      }
    }
  });
}
